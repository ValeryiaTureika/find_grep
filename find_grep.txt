1) find (путь к папке) -name "*n*" - поиск файла по названию, мы точно знаем что в названии есть n, звезды символы до и после
2) find . -name "*n*" - точка поиск во всех нижлежащих папках 
3) find . -type d -name "n*"- поиск директории начинающийся на n. d - искать директорию, f - искать файл
4) grep - поиск информации по текстовому патерну внутри файла 
 grep Anna (путь к файлу с файлом)
5) grep Anna (путь к файлу с файлом) (путь к файлу с файлом) - поиск в двух файлах
6) grep -r Anna . - поиск по файлам строчки Anna
7) grep -r Anna . > Anna_logs.txt (название может быть любое) - сохранить что нашли в файл
8) grep -r -i "anna" . - поиск анна без учета регистра  
9) grep -r -w Anna - поиск строго только Anna, слово целиком, варианты выше пердусматривали, что anna может быть составляющим слова
10) find (путь к папке) | grep Anna * - * поиск во всех файлах 
11)grep -x Anna * - точное совпадение 
12) Создать файл можно также перенаправлением вывода любой команды в файл.
В общем виде команда будет следующей:
какая_нибудь_команда > /путь/к/файлу/наименование_файла
То есть, команда с выводом, затем галочка вправо >, означающая перенаправление вывода команды в файл и путь к файлу.
Если совместить эту команду с mkdir через | без пробелов, выполнится две команды 
mkdir dir_2|ls --help > ls_help.txt
13) grep -l -r sec | xargs -I {} mv {} dir_2/ - найти текстовые файлы у которых в содержимом есть слово “sec”, переместить их в папку
-l - выводить только имена файлов, без результатов поиска 
xargs -I {} - берет данные из стандартного ввода или из файла, разбивает их в соответствии с указанными параметрами, а затем передает другой команде в качестве аргумента, -I опция для запуска нескольких команд с xargs, {} вместилище найденных данных
mv {} dir_2/ - команда, данные, папка
